VideoOutput {
    id: output_%1
    anchors.left: %2
    anchors.top: %3
    width: parent.width/%4
    height: parent.height/%5
    source: videoProducer_%1    

    ButtonDefault {
        id: backBtn_%1
        focus: %6

        anchors {
            margins: base_margins*2
            right: parent.right
            top: parent.top
        }

        class_name: "outline small"
        icon: FontAwesome.icons.fa_arrow_left
        KeyNavigation.down: zoomBtn_%1

        onClicked: {
            wnd.showStartScreen();
            mainView.state = "showTilesView"
        }
    }

    ButtonDefault {
        id: zoomBtn_%1

        anchors {
            margins: base_margins*2
            right: parent.right
            top: backBtn_%1.bottom
        }

        KeyNavigation.down: exitBtn_%1
        class_name: "outline small"
        icon: FontAwesome.icons.fa_search

        onClicked: {
            wnd.hiResUrl = videoProducer_%1.hiResUrl
            wnd.runHiRes();
        }
    }

    ButtonDefault {
        id: exitBtn_%1
        anchors {
            margins: base_margins*2
            right: parent.right
            top: zoomBtn_%1.bottom
        }

        class_name: "outline small"
        icon: FontAwesome.icons.fa_sign_out

        onClicked: {
            Qt.quit()
        }
    }

    Keys.onPressed: {
        if (event.key === Qt.Key_Escape || event.key === Qt.Key_Back) {
            event.accepted = true
            wnd.showStartScreen();
            mainView.state = "showTilesView"
        }
    }

    onVisibleChanged: {
        source.updateTimer(visible)
    }
}
